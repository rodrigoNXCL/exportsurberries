      document.getElementById('loginScreen').style.display = 'none';
      document.getElementById('dashboard').style.display = 'block';

    async function fetchAnalyticsData() {
      try {
        const response = await fetch('/api/analytics-data');
        const result = await response.json();
        
        if (!result.success) throw new Error(result.error);

        updateMetrics(result.data);
        
        document.getElementById('loading').style.display = 'none';
        document.getElementById('content').style.display = 'block';
      } catch (error) {
        console.error('Error:', error);
        document.getElementById('loading').innerHTML = `Error: ${error.message}`;
      }
    }

    function updateMetrics(data) {
      // Actualizar métricas principales
      if (data.main.rows && data.main.rows.length > 0) {
        const lastRow = data.main.rows[data.main.rows.length - 1].metricValues;
        document.getElementById('activeUsers').textContent = formatNumber(lastRow[0].value);
        document.getElementById('pageViews').textContent = formatNumber(lastRow[1].value);
        document.getElementById('sessions').textContent = formatNumber(lastRow[2].value);
        document.getElementById('avgDuration').textContent = formatDuration(lastRow[3].value);
      }

      // Top Páginas
      if (data.pages.rows) {
        const pagesHtml = data.pages.rows.map(row => {
          const views = row.metricValues[0].value;
          return `<div class="data-row"><span>${row.dimensionValues[0].value}</span><span>${formatNumber(views)} vistas</span></div>`;
        }).join('');
        document.getElementById('topPages').innerHTML = pagesHtml;
      }

      // Top Países
      if (data.countries.rows) {
        const countriesHtml = data.countries.rows.map(row => {
          const users = row.metricValues[0].value;
          return `<div class="data-row"><span>${row.dimensionValues[0].value}</span><span>${formatNumber(users)} usuarios</span></div>`;
        }).join('');
        document.getElementById('topCountries').innerHTML = countriesHtml;
      }
    }

    function formatNumber(num) {
      return new Intl.NumberFormat('es-CL').format(num);
    }

    function formatDuration(seconds) {
      const mins = Math.floor(seconds / 60);
      const secs = Math.floor(seconds % 60);
      return `${mins}m ${secs}s`;
    }

    document.getElementById('passwordInput')?.addEventListener('keypress', function(e) {
      if (e.key === 'Enter') window.checkPassword();
    });
  </script>

export const prerender = false;
import { google } from 'googleapis';
import 'dotenv/config';

const GA_PROPERTY_ID = import.meta.env.GA_PROPERTY_ID || '396534500';

export async function GET() {
  try {
    // Crear autenticación OAuth2
    const auth = new google.auth.GoogleAuth({
      credentials: {
        client_id: import.meta.env.GA_CLIENT_ID,
        client_secret: import.meta.env.GA_CLIENT_SECRET,
      },
      scopes: ['https://www.googleapis.com/auth/analytics.readonly'],
    });

    const analyticsData = google.analyticsdata({
      version: 'v1beta',
      auth,
    });

    // Obtener métricas principales (últimos 7 días)
    const response = await analyticsData.properties.runReport({
      property: `properties/${GA_PROPERTY_ID}`,
      dateRanges: [{ startDate: '7daysAgo', endDate: 'today' }],
      dimensions: [{ name: 'date' }],
      metrics: [
        { name: 'activeUsers' },
        { name: 'screenPageViews' },
        { name: 'sessions' },
        { name: 'averageSessionDuration' },
      ],
    });

    // Obtener top páginas
    const pagesResponse = await analyticsData.properties.runReport({
      property: `properties/${GA_PROPERTY_ID}`,
      dateRanges: [{ startDate: '7daysAgo', endDate: 'today' }],
      dimensions: [{ name: 'pageTitle' }, { name: 'pagePath' }],
      metrics: [{ name: 'screenPageViews' }],
      orderBy: [{ metric: { metricName: 'screenPageViews' }, desc: true }],
      limit: 10,
    });

    // Obtener top países
    const countriesResponse = await analyticsData.properties.runReport({
      property: `properties/${GA_PROPERTY_ID}`,
      dateRanges: [{ startDate: '7daysAgo', endDate: 'today' }],
      dimensions: [{ name: 'country' }],
      metrics: [{ name: 'activeUsers' }],
      orderBy: [{ metric: { metricName: 'activeUsers' }, desc: true }],
      limit: 10,
    });

    return new Response(
      JSON.stringify({
        success: true,
        data: {
          main: response,
          pages: pagesResponse,
          countries: countriesResponse,
        },
      }),
      {
        status: 200,
        headers: { 'Content-Type': 'application/json' },
      }
    );
  } catch (error) {
    console.error('Error fetching GA4 data:', error);
    return new Response(
      JSON.stringify({ error: error.message, success: false }),
      { status: 500, headers: { 'Content-Type': 'application/json' } }
    );
  }
}
