    function updateMetrics(data) {
      // Actualizar métricas principales
      if (data.main?.rows?.length > 0) {
        const lastRow = data.main.rows[data.main.rows.length - 1].metricValues;
        document.getElementById('activeUsers').textContent = formatNumber(lastRow[0]?.value || 0);
        document.getElementById('pageViews').textContent = formatNumber(lastRow[1]?.value || 0);
        document.getElementById('sessions').textContent = formatNumber(lastRow[2]?.value || 0);
        document.getElementById('avgDuration').textContent = formatDuration(lastRow[3]?.value || 0);
      }

      // Top Páginas
      if (data.pages?.rows?.length > 0) {
        const pagesHtml = data.pages.rows.map(row => {
          const title = row.dimensionValues?.[0]?.value || 'Sin título';
          const views = row.metricValues?.[0]?.value || 0;
          return `<div class="data-row"><span>${title}</span><span>${formatNumber(views)} vistas</span></div>`;
        }).join('');
        document.getElementById('topPages').innerHTML = pagesHtml;
      } else {
        document.getElementById('topPages').innerHTML = '<p class="no-data">Sin datos de páginas</p>';
      }

      // Top Países
      if (data.countries?.rows?.length > 0) {
        const countriesHtml = data.countries.rows.map(row => {
          const country = row.dimensionValues?.[0]?.value || 'Desconocido';
          const users = row.metricValues?.[0]?.value || 0;
          return `<div class="data-row"><span>${country}</span><span>${formatNumber(users)} usuarios</span></div>`;
        }).join('');
        document.getElementById('topCountries').innerHTML = countriesHtml;
      } else {
        document.getElementById('topCountries').innerHTML = '<p class="no-data">Sin datos de países</p>';
      }
    }

    function formatNumber(num) {
      return new Intl.NumberFormat('es-CL').format(num);
    }

    function formatDuration(seconds) {
      const mins = Math.floor(seconds / 60);
      const secs = Math.floor(seconds % 60);
      return `${mins}m ${secs}s`;
    }

    document.getElementById('passwordInput')?.addEventListener('keypress', function(e) {
      if (e.key === 'Enter') window.checkPassword();
    });

export const prerender = false;

import { getStore } from '@netlify/blobs';
import fs from 'fs/promises';
import path from 'path';
import { fileURLToPath } from 'url';

// Calcular ruta absoluta
const __filename = fileURLToPath(import.meta.url);
const __dirname = path.dirname(__filename);
const ROOT_DIR = path.resolve(__dirname, '..', '..', '..');
const DATA_FILE = path.join(ROOT_DIR, 'public', 'data', 'analytics.json');

export async function POST({ request }) {
  console.log('🚀 [TRACK API] === INICIANDO PETICIÓN ===');
  console.log('[TRACK API] Headers:', Object.fromEntries(request.headers));
  
  try {
    const data = await request.json();
    const { pathname, country = 'Desconocido', countryCode = 'XX', city = 'Desconocida' } = data;
    
    console.log(`📊 [TRACK API] Datos recibidos:`);
    console.log('  - Pathname:', pathname);
    console.log('  - País:', country);
    console.log('  - Ciudad:', city);
    
    // Leer datos existentes
    let analytics = {
      pages: {},
      total: 0,
      dates: {},
      referrers: {},
      countries: {},
      cities: {},
      lastUpdate: null
    };
    
    let usingBlobs = false;
    
    // Intentar Netlify Blobs primero
    try {
      console.log('💾 [TRACK API] Intentando leer de Netlify Blobs...');
      const store = getStore({ name: 'surberries_analytics', consistency: 'strong' });
      const stored = await store.get('data', { type: 'json' });
      
      if (stored) {
        analytics = stored;
        usingBlobs = true;
        console.log('✅ [TRACK API] Datos cargados desde Blobs. Total actual:', analytics.total);
      } else {
        console.log('⚠️ [TRACK API] Blobs está vacío');
      }
    } catch (e) {
      console.log('❌ [TRACK API] Error leyendo Blobs:', e.message);
      console.log('📂 [TRACK API] Usando archivo local como fallback');
    }
    
    // Fallback a archivo local
    if (!usingBlobs) {
      try {
        console.log('📂 [TRACK API] Leyendo archivo local:', DATA_FILE);
        const content = await fs.readFile(DATA_FILE, 'utf-8');
        analytics = JSON.parse(content);
        console.log('✅ [TRACK API] Datos cargados desde archivo. Total:', analytics.total);
      } catch (e) {
        console.log('📄 [TRACK API] Archivo no existe, se creará nuevo');
      }
    }
    
    // Actualizar contadores
    const today = new Date().toISOString().split('T')[0];
    const now = new Date().toISOString();
    
    analytics.total = (analytics.total || 0) + 1;
    analytics.pages[pathname] = (analytics.pages[pathname] || 0) + 1;
    analytics.dates[today] = (analytics.dates[today] || 0) + 1;
    analytics.lastUpdate = now;
    
    if (country && country !== 'Desconocido') {
      analytics.countries = analytics.countries || {};
      analytics.countries[country] = (analytics.countries[country] || 0) + 1;
      
      const cityKey = `${city}, ${countryCode}`;
      analytics.cities = analytics.cities || {};
      analytics.cities[cityKey] = (analytics.cities[cityKey] || 0) + 1;
    }
    
    // Guardar datos
    try {
      if (usingBlobs) {
        const store = getStore({ name: 'surberries_analytics', consistency: 'strong' });
        await store.set('data', JSON.stringify(analytics));
        console.log('💾 [TRACK API] Guardado en Blobs. Nuevo total:', analytics.total);
      } else {
        // Asegurar que el directorio existe
        const dir = path.dirname(DATA_FILE);
        try {
          await fs.mkdir(dir, { recursive: true });
          console.log('📁 [TRACK API] Directorio creado:', dir);
        } catch (e) {
          // Ignorar si ya existe
        }
        
        await fs.writeFile(DATA_FILE, JSON.stringify(analytics, null, 2), 'utf-8');
        console.log('💾 [TRACK API] Guardado en archivo local. Nuevo total:', analytics.total);
        console.log('📍 [TRACK API] Ruta del archivo:', DATA_FILE);
      }
    } catch (e) {
      console.error('❌ [TRACK API] ERROR AL GUARDAR:', e.message);
      console.error('Stack:', e.stack);
      throw e;
    }
    
    console.log('✅ [TRACK API] === PETICIÓN COMPLETADA ===');
    
    return new Response(
      JSON.stringify({ 
        success: true, 
        total: analytics.total,
        message: 'Track registrado correctamente'
      }), 
      {
        status: 200,
        headers: { 
          'Content-Type': 'application/json',
          'Access-Control-Allow-Origin': '*'
        }
      }
    );
    
  } catch (error) {
    console.error('💥 [TRACK API] === ERROR CRÍTICO ===');
    console.error('Error:', error);
    console.error('Message:', error.message);
    console.error('Stack:', error.stack);
    
    return new Response(
      JSON.stringify({ 
        error: error.message,
        success: false 
      }), 
      {
        status: 500,
        headers: { 
          'Content-Type': 'application/json',
          'Access-Control-Allow-Origin': '*'
        }
      }
    );
  }
}

export const prerender = false;
import fs from 'fs/promises';
import path from 'path';
import { fileURLToPath } from 'url';

const __filename = fileURLToPath(import.meta.url);
const __dirname = path.dirname(__filename);
const ROOT_DIR = path.resolve(__dirname, '..', '..', '..');
const DATA_FILE = path.join(ROOT_DIR, 'public', 'data', 'analytics.json');

const ADMIN_PASSWORD = "SurBerries2026!";

export async function POST({ request }) {
  try {
    const authHeader = request.headers.get('Authorization');
    const token = authHeader?.replace('Bearer ', '');
    
    if (token && token !== ADMIN_PASSWORD) {
      return new Response(JSON.stringify({ error: 'Unauthorized' }), { 
        status: 401,
        headers: { 'Content-Type': 'application/json' }
      });
    }

    const empty = {
      pages: {}, 
      total: 0, 
      dates: {}, 
      referrers: {}, 
      countries: {}, 
      cities: {},
      lastUpdate: new Date().toISOString()
    };

    // ✅ BORRAR ARCHIVO LOCAL (funciona en local y producción)
    try {
      const dir = path.dirname(DATA_FILE);
      await fs.mkdir(dir, { recursive: true });
      await fs.writeFile(DATA_FILE, JSON.stringify(empty, null, 2), 'utf-8');
    } catch (e) {
      // Si falla, al menos la API responde correctamente
    }

    return new Response(JSON.stringify({ success: true }), {
      status: 200,
      headers: { 
        'Content-Type': 'application/json',
        'Access-Control-Allow-Origin': '*'
      }
    });
  } catch (error) {
    return new Response(JSON.stringify({ error: error.message }), {
      status: 500,
      headers: { 
        'Content-Type': 'application/json',
        'Access-Control-Allow-Origin': '*'
      }
    });
  }
}
